const mongoose = require('mongoose');

const userSchema = mongoose.Schema({
    UserID: {
        type : String,
        maxlength : 5
        // NOT NULL
    },
    Password:{
        type : String,
        maxlength : 20
        // NOT NULL
    },
    Name : {
        type : String,
        maxlength : 30
    },
    PhoneNum: {
        type : Number,
        maxlength : 11
    },
    Adress: {
        type : String,
        maxlength : 30
        //NOT NULL
    },
    State: {
        type : Number,
        default : 0
        // 0:미승인 1:활동가능 2:정지, NOT NULL
    },
    Term: {
        type : Date
    },
    ReApp: {
        type : Number,
        default : 0
        // 임기 만료시마다 1씩 추가됨. 2이상일시 일반적인 경우 중임 불가, NOT NULL
    }
